\begin{frame}{Third Normal Form (3NF) --- Definition}
  \begin{block}{Definition}
    A relation is in \textbf{3NF} if it is in 2NF and for every non-trivial
    FD $X \to A$, at least one of the following holds:
    \begin{itemize}
      \item $X$ is a \textbf{superkey}, OR
      \item $A$ is a \textbf{prime attribute} (part of some candidate key)
    \end{itemize}
    Informally: \textbf{no non-prime attribute depends on another non-prime attribute}.
  \end{block}

  \begin{block}{Transitive Dependency: The Target}
    A \textbf{transitive dependency} exists when:
    \[
      PK \;\to\; A \;\to\; B
      \quad \text{where } A \text{ and } B \text{ are both non-prime}
    \]
    $B$ does not depend \emph{directly} on the primary key; it reaches
    the PK via the non-prime attribute $A$.
  \end{block}
\end{frame}

% ---------------------------------------------------------------------------
\begin{frame}{3NF --- Identifying the Problem in Our Example}
  After 2NF we have:
  \[
    \texttt{Albums(\underline{AlbumID},\; Title,\; Artist,\; Debut,\; Genre,\; Release)}
  \]

  \vspace{4pt}
  \begin{block}{Functional Dependencies in \texttt{Albums}}
    \begin{itemize}
      \item $\texttt{AlbumID} \to \texttt{Artist}$ \quad (direct)
      \item $\texttt{Artist} \to \texttt{Debut}$ \quad (artist determines their debut year)
      \item Therefore: $\texttt{AlbumID} \to \texttt{Artist} \to \texttt{Debut}$
            \textbf{--- transitive!}
    \end{itemize}
  \end{block}

  \begin{alertblock}{The Problem}
    \texttt{Debut} does not depend directly on \texttt{AlbumID}.  It reaches
    the PK \emph{via} the non-prime attribute \texttt{Artist}.\\[4pt]
    \textbf{Consequence:} If Taylor Swift changes her official debut year
    (e.g.\ corrected biography), we must update \emph{every} row in
    \texttt{Albums} that references her.
  \end{alertblock}
\end{frame}

% ---------------------------------------------------------------------------
\begin{frame}{3NF --- The Fix}
  \textbf{Solution:} Extract the transitive dependency into its own table.

  \vspace{0.5em}
  \begin{columns}[t]
    \begin{column}{0.52\textwidth}
      \textbf{Albums} (\underline{AlbumID})
      \small
      \begin{tabular}{lllll}
        \toprule
        \textbf{AlbumID} & \textbf{Title} & \textbf{ArtistID} & \textbf{Genre} & \textbf{Rel.} \\
        \midrule
        A100 & Midnights & AR01 & Pop & 2022 \\
        A101 & 1989 (TV) & AR01 & Pop & 2023 \\
        \bottomrule
      \end{tabular}
    \end{column}
    \begin{column}{0.42\textwidth}
      \textbf{Artists} (\underline{ArtistID})
      \small
      \begin{tabular}{lll}
        \toprule
        \textbf{ArtistID} & \textbf{Name} & \textbf{Debut} \\
        \midrule
        AR01 & Taylor Swift & 2006 \\
        AR02 & SZA          & 2012 \\
        \bottomrule
      \end{tabular}
    \end{column}
  \end{columns}

  \vspace{0.6em}
  \begin{block}{Final Schema After 3NF}
    \begin{itemize}
      \item \texttt{Artists(\underline{ArtistID}, Name, Debut)}
      \item \texttt{Albums(\underline{AlbumID}, Title, ArtistID\textsuperscript{FK}, Genre, Release)}
      \item \texttt{Songs(\underline{AlbumID, Song}, FeaturedArtist)}
    \end{itemize}
    Debut year is stored \textbf{exactly once} in \texttt{Artists}.
    No anomalies remain.
  \end{block}
\end{frame}

\begin{frame}{Boyce-Codd Normal Form (BCNF) --- Definition}
  \begin{block}{Definition}
    A relation is in \textbf{BCNF} (also: \emph{3.5NF}) if for
    \emph{every} non-trivial FD $X \to A$:
    \[
      X \text{ must be a \textbf{superkey}}
    \]
    No exceptions allowed.
  \end{block}

  \begin{columns}[t]
    \begin{column}{0.5\textwidth}
      \textbf{3NF vs.\ BCNF}
      \begin{itemize}
        \item 3NF \emph{allows} the RHS of an FD to be a prime attribute
              even if the LHS is not a superkey
        \item BCNF \textbf{removes this exception}: every FD must have a
              superkey on the left
        \item BCNF is \textbf{strictly stronger} than 3NF
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{alertblock}{Trade-off}
        BCNF decomposition is always \textbf{lossless} but may \textbf{not
        preserve all FDs}.\\[4pt]
        If dependency-preservation is critical, use 3NF via the Synthesis
        Algorithm instead.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{BCNF --- Classic Example: Overlapping Candidate Keys}
  \begin{block}{Relation \texttt{CourseTeacher}}
    Schema: \texttt{(Student, Course, Teacher)}\\[4pt]
    FDs:
    \begin{itemize}
      \item $\{\texttt{Student, Course}\} \to \texttt{Teacher}$ \quad (one teacher per student per course)
      \item $\texttt{Teacher} \to \texttt{Course}$ \quad (each teacher teaches exactly one course)
    \end{itemize}
    Candidate keys: $\{\texttt{Student, Course}\}$ and $\{\texttt{Student, Teacher}\}$
  \end{block}

  \begin{columns}[t]
    \begin{column}{0.47\textwidth}
      \textbf{Is it 3NF?}\\[2pt]
      FD $\texttt{Teacher} \to \texttt{Course}$:\\
      \texttt{Course} is a prime attribute\\
      (in key $\{\texttt{Student, Course}\}$)\\
      $\Rightarrow$ 3NF \textbf{allows} this.
    \end{column}
    \begin{column}{0.47\textwidth}
      \textbf{Is it BCNF?}\\[2pt]
      FD $\texttt{Teacher} \to \texttt{Course}$:\\
      \texttt{Teacher} alone is \textbf{not} a superkey.\\
      $\Rightarrow$ BCNF \textbf{violation!}
    \end{column}
  \end{columns}

  \vspace{0.4em}
  \begin{block}{BCNF Decomposition}
    $\texttt{TeacherCourse(\underline{Teacher}, Course)}$ \quad and \quad
    $\texttt{StudentTeacher(\underline{Student}, \underline{Teacher})}$
  \end{block}
\end{frame}

\begin{frame}{BCNF --- Music Example: Labels}
  \textbf{Before (3NF but not BCNF):}

  \vspace{4pt}
  \small
  \begin{center}
  \begin{tabular}{llll}
    \toprule
    \textbf{\underline{AlbumID}} & \textbf{\underline{Edition}} & \textbf{LabelID} & \textbf{LabelName} \\
    \midrule
    A100 & 3am Edition        & REP & Republic Records \\
    A101 & Late Night Edition & UMG & Universal Music  \\
    \bottomrule
  \end{tabular}
  \end{center}
  \normalsize

  \vspace{4pt}
  FDs: $\{\texttt{AlbumID, Edition}\} \to \texttt{LabelID} \to \texttt{LabelName}$
  and $\texttt{LabelName} \to \texttt{LabelID}$.\\
  \texttt{LabelID} is \textbf{not} a superkey, yet $\texttt{LabelID} \to \texttt{LabelName}$
  --- \textbf{BCNF violation!}

  \vspace{6pt}
  \textbf{After BCNF decomposition:}

  \begin{columns}[t]
    \begin{column}{0.45\textwidth}
      \small\textbf{Releases (\underline{AlbumID, Edition})}
      \begin{tabular}{lll}
        \toprule
        \textbf{AlbumID} & \textbf{Edition} & \textbf{LabelID} \\
        \midrule
        A100 & 3am Edition        & REP \\
        A101 & Late Night Edition & UMG \\
        \bottomrule
      \end{tabular}
    \end{column}
    \begin{column}{0.38\textwidth}
      \small\textbf{Labels (\underline{LabelID})}
      \begin{tabular}{ll}
        \toprule
        \textbf{LabelID} & \textbf{LabelName} \\
        \midrule
        REP & Republic Records \\
        UMG & Universal Music  \\
        \bottomrule
      \end{tabular}
    \end{column}
  \end{columns}

  \vspace{4pt}
  \normalsize Every determinant is now a superkey. BCNF satisfied. \checkmark
\end{frame}

\begin{frame}{3NF vs.\ BCNF --- Summary}
  \begin{center}
    \small
    \begin{tabular}{lll}
      \toprule
      & \textbf{3NF} & \textbf{BCNF} \\
      \midrule
      Condition         & LHS superkey \emph{or} RHS prime attr. & LHS must be superkey \\
      Strength          & Weaker                                  & Strictly stronger \\
      Lossless?         & Always                                  & Always \\
      Dependency-pres.? & Always (Synthesis Alg.)                 & Not guaranteed \\
      Practical use     & Most production systems                 & When anomalies remain after 3NF \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{block}{Rule of Thumb}
    Start with the \textbf{Synthesis Algorithm} to reach 3NF with all
    dependencies preserved.  Apply BCNF decomposition only if residual anomalies
    exist and losing some FD constraints is acceptable (or can be enforced via
    triggers/application logic).
  \end{block}
\end{frame}
