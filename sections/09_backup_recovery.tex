\section{Backup and Recovery}

\begin{frame}{Why Backup and Recovery?}
  Data is one of the most valuable assets of any organisation. Hardware can be
  replaced in hours; lost data may be gone forever.

  \begin{block}{Two Key Metrics}
    \begin{itemize}
      \item \textbf{RTO -- Recovery Time Objective}: maximum acceptable downtime.\\
            \textit{``How long can we afford to be offline?''}\\
            Example: RTO = 2\,h means the system \textbf{must be back online within 2 hours}.
      \item \textbf{RPO -- Recovery Point Objective}: maximum acceptable data loss in time.\\
            \textit{``How much data can we afford to lose?''}\\
            Example: RPO = 4\,h means we \textbf{accept losing up to 4 hours of transactions}.
    \end{itemize}
  \end{block}

  \begin{examples}{Real-World Examples}
    \begin{tabular}{@{}llll@{}}
      \toprule
      \textbf{System} & \textbf{RTO} & \textbf{RPO} & \textbf{Strategy} \\
      \midrule
      Online bank      & 1\,h    & 0\,min   & Real-time replication \\
      E-commerce shop  & 30\,min & 1\,h     & Hourly log backups \\
      Internal HR tool & 1\,day  & 24\,h    & Nightly full backup \\
      \bottomrule
    \end{tabular}
  \end{examples}
\end{frame}

\begin{frame}{Backup Strategies}
  \begin{block}{Four Types -- From Coarsest to Finest}
    \begin{tabular}{@{}p{2.4cm}p{4.2cm}p{3.5cm}@{}}
      \toprule
      \textbf{Type} & \textbf{What is copied} & \textbf{Restore needs} \\
      \midrule
      \textbf{Full} (weekly) &
        Entire database &
        Only this backup \\[4pt]
      \textbf{Differential} (daily) &
        Changes since last \emph{full} backup &
        Last full + this diff \\[4pt]
      \textbf{Incremental} (hourly) &
        Changes since last backup \emph{of any type} &
        Full + all diffs + all incrementals \\[4pt]
      \textbf{Log backup} (continuous) &
        Transaction log stream &
        Full + logs $\Rightarrow$ point-in-time restore \\
      \bottomrule
    \end{tabular}
  \end{block}

  \begin{examples}{Typical Production Strategy}
    Full backup \textbf{Sunday night} + Differential \textbf{Mon--Sat night} +
    Log backup \textbf{every hour}.\\
    Maximum data loss (RPO): 1 hour. Restore: 2 backups + logs.
  \end{examples}
\end{frame}

\begin{frame}{Recovery -- Failure Scenarios}
  \begin{block}{What Can Go Wrong?}
    \begin{itemize}
      \item \textbf{Software issues}: app bug corrupts data; transaction crashes mid-execution;
            developer runs \texttt{DELETE FROM orders} without \texttt{WHERE}
      \item \textbf{Hardware issues}: sudden power outage, disk head crash, server room flood,
            failed RAID controller
      \item \textbf{Human error}: accidental \texttt{DROP TABLE}, importing bad CSV that
            overwrites good data, wrong rows updated
    \end{itemize}
  \end{block}

  \begin{alertblock}{Classic Horror Story}
    A developer runs \texttt{UPDATE orders SET status = 'shipped'} without a \texttt{WHERE}
    clause in production. All 500,000 orders are now marked as shipped.
    Without a recent backup + transaction log, this data is \textbf{permanently lost}.
  \end{alertblock}
\end{frame}

\begin{frame}{Recovery -- The Three-Step Process}
  \begin{block}{Standard Recovery Process}
    \begin{enumerate}
      \item Restore the most recent \textbf{full backup} -- consistent baseline
      \item Apply the most recent \textbf{differential backup} -- advance to last night
      \item Replay \textbf{transaction log backups} in order up to the recovery point
    \end{enumerate}
  \end{block}

  \begin{block}{Example Timeline}
    \begin{tabular}{@{}ll@{}}
      \toprule
      \textbf{Time} & \textbf{Event} \\
      \midrule
      Sunday 23:00  & Full backup taken (4\,GB) \\
      Monday 23:00  & Differential backup (300\,MB) \\
      Tuesday 10:00--14:00 & Log backups every hour (50\,MB each) \\
      Tuesday 14:37 & \textcolor{red}{\textbf{Crash!}} \\
      Recovery & Full + Diff + Logs $\Rightarrow$ restore to 14:00 \\
      \bottomrule
    \end{tabular}
  \end{block}

\begin{frame}{Transaction Logs: Before- and After-Images}
  Every database engine maintains detailed change logs for every transaction
  to support both \texttt{ROLLBACK} and crash recovery.

  \begin{block}{Before-Image (UNDO log)}
    The state of a record \textbf{before} modification. Written to the log at
    transaction start. Used for \texttt{ROLLBACK} -- restores original data if
    the transaction is aborted.
  \end{block}

  \begin{block}{After-Image (REDO log)}
    The state of a record \textbf{after} modification. Written when the change
    is applied in memory. Used for crash recovery -- if the system crashes after
    \texttt{COMMIT} but before data is flushed to disk, the after-image replays
    the change.
  \end{block}

  \begin{alertblock}{Why both are needed}
    Data changes go to the \textbf{buffer cache} (RAM) first for performance.
    Asynchronous writes to disk happen at \emph{checkpoints}.
    A crash between COMMIT and checkpoint = lost write. After-image fixes this.
    An aborted transaction in RAM = dirty data. Before-image undoes it.
  \end{alertblock}
\end{frame}

\begin{frame}{Transaction Lifecycle with Before- and After-Images}
  Concrete example: a transaction updates the price of ``Cat Milk'' from \EUR{1.99} to \EUR{2.49}.

  \medskip
  \begin{enumerate}
    \item \textbf{START TRANSACTION}: the DBMS begins tracking this transaction.
    \item \textbf{Create Before-Image}: the current record (price = 1.99) is written to the
          transaction log as the before-image.
    \item \textbf{Modify data in cache}: the price is updated to 2.49 in the in-memory buffer.
          The physical disk file is \emph{not yet changed}.
    \item \textbf{Create After-Image}: the updated record (price = 2.49) is written to the
          transaction log as the after-image.
    \item \textbf{COMMIT or ROLLBACK}:
      \begin{itemize}
        \item \textbf{COMMIT}: the change is confirmed. The before-image is deleted from the log.
              The after-image remains until the buffer is flushed to disk at the next checkpoint,
              then it too is deleted.
        \item \textbf{ROLLBACK}: the before-image is used to restore the record to price = 1.99.
              Both before- and after-image are discarded.
      \end{itemize}
  \end{enumerate}

  \medskip
  \begin{block}{Why this matters}
    This mechanism directly implements two ACID properties: \textbf{Atomicity} (all-or-nothing via
    rollback) and \textbf{Durability} (committed changes survive crashes via redo from the log).
  \end{block}
\end{frame}

\begin{frame}{RAID -- Redundant Array of Independent Disks}
  \begin{block}{What Is RAID?}
    \textbf{RAID} combines multiple physical hard drives into a single logical
    storage unit, providing improved \textbf{performance} (via parallelism)
    and/or \textbf{fault tolerance} (via redundancy).
  \end{block}

  \begin{block}{Three Fundamental Techniques}
    \begin{itemize}
      \item \textbf{Striping}: data split into blocks, written across disks in parallel.
            Increases read/write throughput.
      \item \textbf{Mirroring}: exact copy maintained on a second disk.
            Instant failover if one disk fails.
      \item \textbf{Parity}: mathematical checksum stored alongside data.
            Allows reconstruction of a failed disk without a full mirror.
    \end{itemize}
  \end{block}

  \begin{alertblock}{No Silver Bullet}
    There is always a trade-off between performance, redundancy, and storage
    efficiency. No single RAID level maximises all three.
  \end{alertblock}
\end{frame}

\begin{frame}{RAID Level Comparison}
  \begin{center}
  \renewcommand{\arraystretch}{1.3}
  \small
  \begin{tabular}{@{}lllll@{}}
    \toprule
    \textbf{Level} & \textbf{Techniques} & \textbf{Min Disks} & \textbf{Storage Eff.} & \textbf{Fault Tolerance} \\
    \midrule
    \textbf{RAID 0} & Striping only     & 2 & 100\% & None (0 failures) \\
    \textbf{RAID 1} & Mirroring only    & 2 & 50\%  & 1 disk failure \\
    \textbf{RAID 5} & Stripe + Parity   & 3 & $(n-1)/n$ & 1 disk failure \\
    \textbf{RAID 6} & Stripe + 2 Parity & 4 & $(n-2)/n$ & 2 disk failures \\
    \textbf{RAID 10}& Mirror + Stripe   & 4 & 50\%  & 1 per mirrored pair \\
    \bottomrule
  \end{tabular}
  \end{center}

  \begin{block}{Quick Guide for Databases}
    \begin{itemize}
      \item \textbf{RAID 0}: never for databases (zero redundancy)
      \item \textbf{RAID 1}: OS drives, small critical databases
      \item \textbf{RAID 5}: most common for NAS/SAN storage arrays
      \item \textbf{RAID 10}: high-performance database servers (best of RAID 1+0)
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{RAID 0 -- Striping}
  \begin{columns}
    \begin{column}{0.65\textwidth}
      \textbf{How it works:} data is split into equal-sized blocks (\emph{stripes}) and written
      across all disks in round-robin order.

      \medskip
      Example with 2 disks: Disk\,0 receives blocks A1, A2, A3, A4; Disk\,1 receives A5, A6, A7,
      A8. Both disks are written simultaneously.

      \medskip
      \textbf{Pros:}
      \begin{itemize}
        \item \textbf{Fastest read and write performance} of all RAID levels (full parallelism).
        \item \textbf{100\% storage efficiency} -- no space wasted on redundancy.
      \end{itemize}

      \textbf{Cons:}
      \begin{itemize}
        \item \textbf{Zero fault tolerance}: if even \emph{one} disk fails, all data is lost
              (stripes from the failed disk are missing and cannot be reconstructed).
        \item RAID 0 is technically \emph{not} redundant despite being called RAID.
      \end{itemize}

      \begin{alertblock}{Use case (!)}
        Temporary scratch storage, video editing caches, or any scenario where speed matters and
        data loss is acceptable. Never use for production databases.
      \end{alertblock}
    \end{column}
    \begin{column}{0.3\textwidth}
      \centering
      \textbf{Disk 0} \quad \textbf{Disk 1}\\[4pt]
      A1 \quad A5\\
      A2 \quad A6\\
      A3 \quad A7\\
      A4 \quad A8\\
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{RAID 1 -- Mirroring}
  \begin{columns}
    \begin{column}{0.65\textwidth}
      \textbf{How it works:} every write operation is duplicated to two (or more) disks
      simultaneously. Disk\,0 and Disk\,1 are always byte-for-byte identical.

      \medskip
      Example with 2 disks: Disk\,0 = A1, A2, A3, A4 and Disk\,1 = A1, A2, A3, A4.

      \medskip
      \textbf{Pros:}
      \begin{itemize}
        \item \textbf{Full redundancy}: if one disk fails, the mirror takes over instantly with no
              data loss and no rebuild time.
        \item \textbf{Fast reads}: the controller can serve read requests from either disk in
              parallel, doubling read throughput.
      \end{itemize}

      \textbf{Cons:}
      \begin{itemize}
        \item \textbf{50\% storage efficiency}: you pay for two disks but only get the capacity of
              one.
        \item Most expensive per usable gigabyte of all RAID levels.
      \end{itemize}

      \begin{examples}{Use case (!)}
        Database servers, operating system drives, any storage where uptime and data availability
        are the top priority.
      \end{examples}
    \end{column}
    \begin{column}{0.3\textwidth}
      \centering
      \textbf{Disk 0} \quad \textbf{Disk 1}\\[4pt]
      A1 \quad A1\\
      A2 \quad A2\\
      A3 \quad A3\\
      A4 \quad A4\\
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{RAID 5 -- Striping with Distributed Parity}
  \textbf{How it works:} data is striped across $n \geq 3$ disks. For each stripe, a \emph{parity
  block} is computed and stored on one of the disks (the parity location rotates across disks so no
  single disk becomes the bottleneck).

  \medskip
  If one disk fails, its data can be \textbf{mathematically reconstructed} from the remaining
  disks and the parity blocks using XOR operations.

  \medskip
  \textbf{Pros:}
  \begin{itemize}
    \item Good balance of \textbf{performance}, \textbf{storage efficiency}, and
          \textbf{fault tolerance}.
    \item Storage efficiency: $\frac{n-1}{n}$ -- with 4 disks, 75\% of total capacity is usable.
    \item Survives the failure of \textbf{exactly one disk} without data loss.
  \end{itemize}

  \textbf{Cons:}
  \begin{itemize}
    \item \textbf{Slower writes} than RAID 0 or RAID 1 (parity must be computed for every write).
    \item If \textbf{two disks fail simultaneously}, all data is lost (no second parity).
    \item After a disk failure, \textbf{rebuild time} can take hours or days on large arrays --
          during which the remaining disks are under extra stress.
  \end{itemize}

  \begin{block}{Use case (!)}
    The most common RAID level in enterprise NAS and SAN storage arrays where a balance of cost,
    performance, and safety is needed.
  \end{block}
\end{frame}
