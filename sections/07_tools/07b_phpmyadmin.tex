\begin{frame}{phpMyAdmin -- Overview}
  \begin{block}{What Is phpMyAdmin?}
    \textbf{phpMyAdmin} is a free, browser-based administration tool for MySQL
    and MariaDB. It ships pre-installed with XAMPP -- no extra setup required.\\[0.4em]
    Access at: \texttt{http://localhost/phpmyadmin} (after starting MySQL in XAMPP)
  \end{block}

  \begin{block}{Core Capabilities}
    \begin{itemize}
      \item Create, modify, and drop \textbf{databases} and \textbf{tables}
      \item Execute \textbf{SQL queries} directly in the browser
      \item \textbf{Import} (CSV, SQL dump) and \textbf{export} in multiple formats
      \item Manage \textbf{users and permissions}
      \item Browse and edit table contents visually (no SQL needed)
    \end{itemize}
  \end{block}

  \begin{alertblock}{When to use phpMyAdmin vs.\ DBeaver}
    \begin{tabular}{@{}ll@{}}
      \textbf{phpMyAdmin}  & Quick admin tasks, import/export, user management \\
      \textbf{DBeaver}     & Writing and testing SQL, handling large result sets \\
    \end{tabular}
  \end{alertblock}
\end{frame}

\begin{frame}{phpMyAdmin -- UI Layout}
  \begin{block}{Two-Panel Interface}
    \begin{itemize}
      \item \textbf{Left panel}: collapsible tree of databases and tables
      \item \textbf{Right panel}: active work area
        \begin{itemize}
          \item \textit{Structure} tab: columns, types, indexes, constraints
          \item \textit{SQL} tab: free-form SQL editor with syntax highlighting
          \item \textit{Insert} tab: form-based row insertion
          \item \textit{Export/Import} tabs: dump and restore data
        \end{itemize}
    \end{itemize}
  \end{block}

  \begin{examples}{Typical Workflow for Course Exercises}
    \begin{enumerate}
      \item Select your database in the left panel
      \item Click the \textbf{SQL} tab
      \item Type your query and press \textbf{Go}
      \item Results appear as a table below the editor
    \end{enumerate}
  \end{examples}
\end{frame}
