\begin{frame}{DBeaver -- Universal Database Client}
  \begin{block}{What Is DBeaver?}
    \textbf{DBeaver} is a free, open-source, cross-platform database client that supports
    virtually every database system via JDBC drivers: MySQL, MariaDB, PostgreSQL,
    SQLite, Oracle, SQL Server, MongoDB, and many more.\\[0.3em]
    Download: \texttt{dbeaver.io} \quad (Community Edition is fully free)
  \end{block}

  \begin{block}{Key Features}
    \begin{itemize}
      \item \textbf{Universal connectivity}: one tool for all databases
      \item \textbf{SQL editor}: syntax highlighting, auto-completion, query history
      \item \textbf{ER diagram viewer}: reverse-engineers schema into ERM automatically
      \item \textbf{Data editor}: browse/edit table contents with filtering and sorting
      \item \textbf{Export/import}: CSV, JSON, SQL dump, Excel -- all built in
      \item \textbf{EXPLAIN viewer}: graphical execution plan for query profiling
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{DBeaver -- Connecting to MySQL (XAMPP)}
  \begin{block}{Setting Up a Connection}
    \begin{enumerate}
      \item Open DBeaver $\to$ click \textbf{``New Database Connection''} (plug icon)
      \item Select \textbf{MySQL} from the driver list $\to$ Next
      \item Enter connection details:
        \begin{itemize}
          \item \textbf{Host}: \texttt{localhost}
          \item \textbf{Port}: \texttt{3306}
          \item \textbf{Username}: \texttt{root}
          \item \textbf{Password}: \textit{(empty for XAMPP default)}
        \end{itemize}
      \item Click \textbf{Test Connection} -- DBeaver downloads the JDBC driver
            automatically if not yet installed
      \item Click \textbf{Finish}
    \end{enumerate}
  \end{block}

  \begin{examples}{Why DBeaver over phpMyAdmin?}
    DBeaver is a \emph{native desktop application}: faster, supports keyboard
    shortcuts, handles large result sets better, and works offline without a
    running Apache web server. Preferred by most developers for daily SQL work.
  \end{examples}
\end{frame}
