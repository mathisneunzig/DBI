\begin{frame}
  \frametitle{MySQL Workbench}

  \begin{block}{What Is MySQL Workbench?}
    \textbf{MySQL Workbench} is the official, free GUI tool from Oracle for MySQL databases.\\
    Download: \texttt{dev.mysql.com/downloads/workbench/}\\[0.3em]
    It combines a SQL editor, visual schema designer, and server administration panel in one application.
  \end{block}

  \begin{block}{Key Features}
    \begin{itemize}
      \item \textbf{Visual schema design}: drag-and-drop ERM / schema diagrams
      \item \textbf{SQL editor}: auto-completion, syntax highlighting, query history
      \item \textbf{Server administration}: user management, performance monitoring, log viewer
      \item \textbf{Data migration wizard}: migrate data from other DBMS platforms
      \item \textbf{Forward engineering}: generate \texttt{CREATE} SQL scripts directly from your ERM diagram
      \item \textbf{Reverse engineering}: auto-generate an ERM diagram from an existing live database
    \end{itemize}
  \end{block}

  \begin{examples}{In This Course}
    MySQL Workbench is used for \textbf{designing and documenting database schemas}. The forward-engineering workflow -- draw the ERM, then generate the DDL -- mirrors real-world database development practice.
  \end{examples}
