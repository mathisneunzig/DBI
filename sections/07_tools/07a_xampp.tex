\section{Tools for Database Development}

% ------------------------------------------------------------
\begin{frame}{XAMPP -- Overview}
  \begin{block}{What Is XAMPP?}
    \textbf{XAMPP} is a free, open-source software bundle that installs a fully
    functional local web server and database environment in minutes.
    \begin{itemize}
      \item Developed by \textbf{Apache Friends} since 2002
      \item Available for \textbf{Windows}, \textbf{macOS}, and \textbf{Linux}
            (hence ``cross-platform'' = X)
      \item Used in this course to host a local \textbf{MySQL\,/\,MariaDB} database
    \end{itemize}
  \end{block}

  \begin{block}{The Acronym Explained}
    \centering
    \begin{tabular}{@{}cl@{}}
      \textbf{X} & Cross-platform \\
      \textbf{A} & Apache HTTP Server \\
      \textbf{M} & MariaDB (originally MySQL) \\
      \textbf{P} & PHP \\
      \textbf{P} & Perl \\
    \end{tabular}
  \end{block}

  \begin{alertblock}{Why XAMPP for this course?}
    Zero configuration -- every component works out of the box. Ideal for
    local development where setting up a production server would be overkill.
  \end{alertblock}
\end{frame}

% ------------------------------------------------------------
\begin{frame}{XAMPP -- Installation}
  \begin{block}{Download \& Install}
    \begin{enumerate}
      \item Download the installer from \textbf{apachefriends.org}
      \item Run the installer -- \textit{accepting default paths is strongly recommended}
      \item Default installation directories:
        \begin{itemize}
          \item Windows: \texttt{C:\textbackslash xampp}
          \item macOS: \texttt{/Applications/XAMPP}
        \end{itemize}
    \end{enumerate}
  \end{block}

  \begin{block}{After Installation: The Control Panel}
    The \textbf{XAMPP Control Panel} is your central dashboard:
    \begin{itemize}
      \item Lists all bundled services: Apache, \textbf{MySQL}, FileZilla, Mercury
      \item Click \textbf{``Start''} next to \textit{MySQL} to start the database
      \item Status indicator turns \textcolor{green!60!black}{\textbf{green}} when running
    \end{itemize}
  \end{block}

  \begin{examples}{Common First-Time Pitfall on Windows}
    Port conflicts with Skype or IIS. If MySQL fails to start, check whether
    another process occupies port \textbf{3306}. Fix: change XAMPP's port in
    \texttt{my.ini} or stop the conflicting service.
  \end{examples}
\end{frame}

\begin{frame}{XAMPP -- Starting and Accessing MySQL}
  \begin{block}{Default Credentials}
    \begin{tabular}{@{}ll@{}}
      \textbf{Host}     & \texttt{localhost} \\
      \textbf{Port}     & \texttt{3306} \\
      \textbf{Username} & \texttt{root} \\
      \textbf{Password} & \textit{(empty by default)} \\
    \end{tabular}
  \end{block}

  \begin{alertblock}{Security Warning}
    An empty root password is acceptable \textit{only} on a local machine with
    no external access. \textbf{Never} deploy XAMPP with default credentials
    in a production environment!
  \end{alertblock}

  \begin{block}{Access Options}
    \begin{itemize}
      \item \textbf{phpMyAdmin} (browser): \texttt{http://localhost/phpmyadmin}
      \item \textbf{MySQL command line}: available via the XAMPP shell button
      \item \textbf{External SQL clients}: MySQL Workbench, DBeaver --
            connect to \texttt{localhost:3306}
    \end{itemize}
    \textit{For this course: use phpMyAdmin (quick tasks) or DBeaver (SQL work).}
  \end{block}
\end{frame}
