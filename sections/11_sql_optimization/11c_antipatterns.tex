\begin{frame}[fragile]{Common Query Anti-Patterns}
  \begin{block}{Patterns That Prevent Index Use or Waste Resources}
\begin{lstlisting}[style=sqlstyle]
-- Anti-pattern 1: function on indexed column (index bypassed!)
WHERE UPPER(Name) = 'THE BEATLES'   -- index on Name not used
-- Fix: store data in normalised case, or use a functional index

-- Anti-pattern 2: leading wildcard (full table scan)
WHERE Title LIKE '%Road%'            -- cannot use B-tree index
-- Fix: use FULLTEXT index for free-text search

-- Anti-pattern 3: implicit type conversion (may skip index)
WHERE ArtistID = '1'  -- ArtistID is INT, '1' is VARCHAR
-- Fix: match data type: WHERE ArtistID = 1

-- Anti-pattern 4: SELECT * (pulls every column every row)
SELECT * FROM Album;
-- Fix: name only the columns you actually need

-- Anti-pattern 5: correlated subquery in SELECT
SELECT Name,
  (SELECT COUNT(*) FROM Album WHERE ArtistID = ar.ArtistID)
FROM Artist ar;   -- re-executes subquery for EVERY artist row!
-- Fix: use LEFT JOIN + GROUP BY
\end{lstlisting}
  \end{block}
\end{frame}

\begin{frame}[fragile]{The N+1 Query Problem}
  The \textbf{N+1 problem} is one of the most common performance pitfalls
  in applications that use ORMs or generate SQL in loops.

  \begin{block}{What it is -- with Music DB}
\begin{lstlisting}[style=sqlstyle]
-- Step 1: one query to fetch all albums (returns N rows)
SELECT * FROM Album;   -- returns 3 albums

-- Step 2: N queries, one per album, to get the artist name
SELECT Name FROM Artist WHERE ArtistID = 1;  -- for Abbey Road
SELECT Name FROM Artist WHERE ArtistID = 2;  -- for Autobahn
SELECT Name FROM Artist WHERE ArtistID = 3;  -- for RAM
-- = 3 extra queries (N=3) => 1 + 3 = 4 total queries
\end{lstlisting}
  \end{block}

  \begin{block}{The Fix: One JOIN Instead of N+1 Queries}
\begin{lstlisting}[style=sqlstyle]
-- Single query returns everything needed
SELECT al.Title, ar.Name AS artist_name, al.ReleaseYear
FROM   Album  al
JOIN   Artist ar ON al.ArtistID = ar.ArtistID;
\end{lstlisting}
  \end{block}

  \begin{alertblock}{Why It Matters at Scale}
    With 10,000 albums: N+1 generates \textbf{10,001 database round-trips}.
    Even at 1\,ms per query = \textbf{10 seconds}.
    A single JOIN returns the same data in \textbf{milliseconds}.
  \end{alertblock}
\end{frame}

\begin{frame}[fragile]{Query Optimization -- Practical Checklist}
  \begin{block}{Step-by-Step Approach}
    \begin{enumerate}
      \item \textbf{Run \texttt{EXPLAIN ANALYZE}} -- find the slowest operation
            (\texttt{type=ALL}, high \texttt{rows}, \texttt{Using filesort})
      \item \textbf{Check for missing indexes} on \texttt{WHERE},
            \texttt{JOIN ON}, and \texttt{ORDER BY} columns
      \item \textbf{Avoid functions on indexed columns} in \texttt{WHERE} --
            rewrite the predicate instead
      \item \textbf{Select only needed columns} -- drop \texttt{SELECT *};
            use covering indexes where possible
      \item \textbf{Push filters early} -- filter rows close to the source
            table, before joins
      \item \textbf{Replace correlated subqueries} with \texttt{JOIN} or
            window functions
      \item \textbf{Batch writes}: wrap many small \texttt{INSERT}s in one
            multi-row \texttt{INSERT} or a transaction
      \item \textbf{Update statistics}: run \texttt{ANALYZE TABLE} so the
            planner has fresh data
    \end{enumerate}
  \end{block}
\end{frame}
