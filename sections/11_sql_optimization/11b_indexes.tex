\begin{frame}{Indexes -- What They Are}
  An \textbf{index} is a separate data structure maintained alongside a table.
  It maps column values to physical row locations, enabling fast lookups
  without scanning every row.

  \begin{block}{Analogy}
    The index at the back of a textbook. Instead of reading every page to find
    ``normalization'', you look up the word in the index and jump to the right
    page. A database index works the same way.
  \end{block}

  \begin{block}{Trade-offs}
    \begin{tabular}{@{}ll@{}}
      \toprule
      \textbf{Benefit} & \textbf{Cost} \\
      \midrule
      Fast \texttt{WHERE} lookups              & Extra disk space \\
      Fast \texttt{JOIN} on foreign keys       & Slower \texttt{INSERT/UPDATE/DELETE} \\
      Fast \texttt{ORDER BY / GROUP BY}        & DBMS must keep index up to date \\
      \bottomrule
    \end{tabular}
  \end{block}

  \begin{alertblock}{Primary and Unique Keys Are Automatically Indexed}
    MySQL always creates a \textbf{clustered B-tree index} on the primary key.
    \texttt{UNIQUE} constraints also create an index automatically.
  \end{alertblock}
\end{frame}

% ------------------------------------------------------------
\begin{frame}{B-Tree Index -- How It Works}
  The default index type in MySQL and most relational DBMSs is a
  \textbf{B-Tree (Balanced Tree)} index.

  \begin{block}{Structure}
    \begin{itemize}
      \item The tree is sorted by the indexed column value
      \item \textbf{Root} $\to$ \textbf{internal nodes} narrow the search range
      \item \textbf{Leaf nodes} store the indexed values + pointers to rows
      \item Always balanced: every leaf is at the same depth,
            so lookup time is $O(\log n)$ regardless of which value is searched
    \end{itemize}
  \end{block}

  \begin{block}{What B-Tree Indexes Support}
    \begin{tabular}{@{}ll@{}}
      \toprule
      \textbf{Query type} & \textbf{Example} \\
      \midrule
      Equality   & \texttt{WHERE Genre = 'Rock'} \\
      Range      & \texttt{WHERE ReleaseYear BETWEEN 1970 AND 1980} \\
      Prefix     & \texttt{WHERE Title LIKE 'Abbey\%'} \\
      Sorting    & \texttt{ORDER BY} on indexed column -- skips filesort \\
      \bottomrule
    \end{tabular}
  \end{block}
\end{frame}

% ------------------------------------------------------------
\begin{frame}[fragile]{Creating and Dropping Indexes -- Music DB}
  \begin{block}{Syntax}
\begin{lstlisting}[style=sqlstyle]
-- Index for "find tracks by album" queries
CREATE INDEX idx_track_album ON Track(AlbumID);

-- Index for "find albums by artist and year"
CREATE INDEX idx_album_artist_year
    ON Album(ArtistID, ReleaseYear);

-- Unique index -- also enforces uniqueness constraint
CREATE UNIQUE INDEX uq_artist_name ON Artist(Name);

-- Show all indexes on a table
SHOW INDEX FROM Track;

-- Remove an index
DROP INDEX idx_track_album ON Track;
\end{lstlisting}
  \end{block}

  \begin{block}{When the Planner Uses the Index}
    The planner uses an index when it estimates an index scan is cheaper than
    a full table scan. For very small tables (a few hundred rows), a full scan
    is often faster -- the planner will choose accordingly.
  \end{block}
\end{frame}

% ------------------------------------------------------------
\begin{frame}[fragile]{Index Selection Guide}
  \begin{block}{When to Add an Index}
\begin{lstlisting}[style=sqlstyle]
-- Rule 1: Index every foreign key column
CREATE INDEX idx_album_artist ON Album(ArtistID);
CREATE INDEX idx_track_album  ON Track(AlbumID);

-- Rule 2: Index columns frequently used in WHERE
CREATE INDEX idx_album_genre ON Album(Genre);

-- Rule 3: Index columns used in ORDER BY (avoids filesort)
CREATE INDEX idx_album_year ON Album(ReleaseYear);

-- Rule 4: Composite index -- most selective column first
-- Query: WHERE ArtistID = 3 AND Genre = 'Electronic'
CREATE INDEX idx_album_artist_genre ON Album(ArtistID, Genre);
\end{lstlisting}
  \end{block}

  \begin{alertblock}{Do NOT Index Everything}
    Every index slows down \texttt{INSERT/UPDATE/DELETE}.
    A table with 10 indexes takes $\sim$10x longer to write to.
    Add indexes only where \texttt{EXPLAIN} shows a problem.
  \end{alertblock}
\end{frame}

% ------------------------------------------------------------
\begin{frame}{Composite Indexes and Column Order}
  A \textbf{composite index} covers multiple columns.
  The column order matters enormously.

  \begin{block}{The Leftmost Prefix Rule}
    Index on \texttt{(ArtistID, Genre)} can be used for:
    \begin{itemize}
      \item \texttt{WHERE ArtistID = ?} \quad -- uses index (leftmost col)
      \item \texttt{WHERE ArtistID = ? AND Genre = ?} \quad -- uses full index
      \item \texttt{WHERE Genre = ?} \quad -- \textbf{does NOT use index}
            (Genre is not the leftmost column)
      \item \texttt{WHERE ArtistID = ? AND ReleaseYear = ?} \quad
            -- uses index on ArtistID only, filters ReleaseYear manually
    \end{itemize}
  \end{block}

  \begin{examples}{Design Tip}
    Put the most selective (highest cardinality) column first.
    If queries filter first by \texttt{ArtistID} then optionally by \texttt{Genre},
    index should be \texttt{(ArtistID, Genre)}, not \texttt{(Genre, ArtistID)}.
  \end{examples}
\end{frame}

% ------------------------------------------------------------
\begin{frame}{Covering Indexes}
  A \textbf{covering index} contains \emph{all} columns a query needs --
  the DBMS answers the query entirely from the index, without touching the
  actual table rows.

  \begin{block}{Music Database Example}
    Query: \texttt{SELECT Genre FROM Album WHERE ArtistID = 2;}

    \medskip
    If the index is \texttt{(ArtistID, Genre)}, the engine finds the matching
    \texttt{ArtistID} entries in the index and reads \texttt{Genre} directly
    from the index leaf nodes. \textbf{No row lookup required.}
  \end{block}

  \begin{block}{When Does It Help Most?}
    \begin{itemize}
      \item The table is very wide (many columns) but the query needs only a few
      \item The table has many rows and the query runs frequently
      \item EXPLAIN shows \texttt{Extra: Using index} when a covering index is used
    \end{itemize}
  \end{block}
\end{frame}

% ------------------------------------------------------------
\begin{frame}[fragile]{When NOT to Use an Index}
  \begin{block}{Situations Where Indexes Hurt More Than They Help}
    \begin{itemize}
      \item \textbf{Very small tables}: a full scan of 500 rows is faster
            than an index lookup + row fetch
      \item \textbf{Low-cardinality columns}: a \texttt{BOOLEAN} or a
            \texttt{Gender} column with 2--3 distinct values. The index
            barely reduces rows scanned
      \item \textbf{Write-heavy columns}: every \texttt{UPDATE/INSERT/DELETE}
            must update the index too. Too many indexes slow down writes
            significantly
      \item \textbf{Functions on indexed columns}: defeats the index
    \end{itemize}
  \end{block}

  \begin{alertblock}{Function on Indexed Column -- Classic Mistake}
\begin{lstlisting}[style=sqlstyle]
-- WRONG: index on ReleaseYear is NOT used
WHERE YEAR(ReleaseYear) = 1969

-- CORRECT: index IS used (range query on indexed column)
WHERE ReleaseYear >= 1969 AND ReleaseYear < 1970
\end{lstlisting}
  \end{alertblock}
\end{frame}
