\subsection{Data Control Language}

% ----------------------------------------------------------
\begin{frame}{DCL -- Data Control Language}
  \textbf{DCL} controls \emph{who} can do \emph{what} in a database.
  Two core statements: \texttt{GRANT} (give permissions) and \texttt{REVOKE} (take them away).

  \begin{block}{Privilege Types in MySQL}
    \begin{columns}[T]
      \column{0.48\linewidth}
        \textbf{DML / Query privileges:}
        \begin{itemize}
          \item \texttt{SELECT}
          \item \texttt{INSERT}
          \item \texttt{UPDATE}
          \item \texttt{DELETE}
        \end{itemize}
      \column{0.48\linewidth}
        \textbf{Schema / Admin privileges:}
        \begin{itemize}
          \item \texttt{CREATE}, \texttt{DROP}
          \item \texttt{ALTER}, \texttt{INDEX}
          \item \texttt{GRANT OPTION}
          \item \texttt{ALL PRIVILEGES}
        \end{itemize}
    \end{columns}
  \end{block}

  \begin{block}{Principle of Least Privilege}
    Grant only the \textbf{minimum permissions} a user or application actually needs.
    Never run application code as \texttt{root}. A compromised application account
    with only \texttt{SELECT} cannot delete or modify data.
  \end{block}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{GRANT -- Giving Permissions}
  \begin{lstlisting}[style=sqlstyle]
-- Create a user first
CREATE USER 'app_reader'@'localhost' IDENTIFIED BY 's3cr3t';

-- Grant SELECT on the entire music_db database
GRANT SELECT ON music_db.* TO 'app_reader'@'localhost';

-- Grant specific privileges on a single table
GRANT INSERT, UPDATE ON music_db.Track
    TO 'app_editor'@'localhost';

-- Grant ALL privileges (admin -- use with extreme care!)
GRANT ALL PRIVILEGES ON music_db.*
    TO 'db_admin'@'localhost'
    WITH GRANT OPTION;  -- can pass privileges on to others

-- Apply privilege changes immediately
FLUSH PRIVILEGES;
  \end{lstlisting}

  \begin{itemize}
    \item \texttt{'user'@'\%'}: connects from \textbf{any host}.
    \item \texttt{'user'@'localhost'}: connects from \textbf{same machine only} (more secure).
  \end{itemize}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{REVOKE -- Removing Permissions}
  \begin{lstlisting}[style=sqlstyle]
-- Remove INSERT from a specific table
REVOKE INSERT ON music_db.Track
    FROM 'app_editor'@'localhost';

-- Remove all privileges from the database
REVOKE ALL PRIVILEGES ON music_db.*
    FROM 'app_reader'@'localhost';

-- Show what privileges a user currently has
SHOW GRANTS FOR 'app_reader'@'localhost';

-- Remove a user entirely
DROP USER 'app_reader'@'localhost';
  \end{lstlisting}

  \begin{alertblock}{REVOKE Does Not Cascade}
    If user A granted privileges to user B using \texttt{WITH GRANT OPTION}, revoking
    A's privileges does \textbf{not} automatically revoke B's derived privileges in MySQL.
    You must revoke from each user individually.
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{Role-Based Access Control (RBAC)}
  Managing privileges per-user becomes unmanageable at scale.
  \textbf{Roles} group privileges and are then assigned to users.

  \begin{lstlisting}[style=sqlstyle]
-- Create roles (MySQL 8+)
CREATE ROLE 'music_readonly', 'music_editor', 'music_admin';

-- Assign privileges to roles
GRANT SELECT ON music_db.* TO 'music_readonly';
GRANT SELECT, INSERT, UPDATE, DELETE ON music_db.*
    TO 'music_editor';
GRANT ALL PRIVILEGES ON music_db.* TO 'music_admin';

-- Assign roles to users
GRANT 'music_readonly' TO 'alice'@'localhost';
GRANT 'music_editor'   TO 'bob'@'localhost';
GRANT 'music_admin'    TO 'carol'@'localhost';

-- Set default role so it activates at login
SET DEFAULT ROLE 'music_readonly' TO 'alice'@'localhost';
  \end{lstlisting}
\end{frame}

    \draw[arrow] (bob)   -- (edit);
    \draw[arrow] (carol) -- (adm);

    \draw[arrow] (edit) -- (p2);
    \draw[arrow] (adm)  -- (p3);
  \end{tikzpicture}
  \end{center}

  \begin{block}{RBAC Advantages}
    When a user's responsibilities change, update one \texttt{GRANT} assignment -- not
    dozens of individual privilege statements. Roles can also be nested.
  \end{block}
\end{frame}
