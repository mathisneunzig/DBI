\begin{frame}[fragile]{ORDER BY -- Sorting Results}
  \begin{block}{Sorting Query Results -- Music Database}
\begin{lstlisting}[style=sqlstyle]
-- Tracks sorted by duration, shortest first (default: ASC)
SELECT Title, Duration FROM Track ORDER BY Duration;

-- Albums sorted newest first
SELECT Title, ReleaseYear FROM Album ORDER BY ReleaseYear DESC;

-- Multiple columns: primary then secondary sort
SELECT Title, Genre, ReleaseYear FROM Album
ORDER BY Genre ASC, ReleaseYear DESC;
\end{lstlisting}
  \end{block}

  \begin{block}{Result of \texttt{ORDER BY ReleaseYear DESC}}
    \begin{tabular}{@{}ll@{}}
      \toprule
      \textbf{Title} & \textbf{ReleaseYear} \\
      \midrule
      Random Access Memories & 2013 \\
      Autobahn & 1974 \\
      Abbey Road & 1969 \\
      \bottomrule
    \end{tabular}
  \end{block}

  \begin{itemize}
    \item Without \texttt{ORDER BY}, row order is \textbf{undefined} and can vary
    \item \texttt{NULL} values sort to the \emph{end} in MySQL (\texttt{ASC} direction)
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{GROUP BY -- Grouping Rows}
  \begin{block}{Computing Per-Group Statistics}
\begin{lstlisting}[style=sqlstyle]
-- How many tracks does each album have?
SELECT AlbumID, COUNT(*) AS track_count
FROM Track
GROUP BY AlbumID;
\end{lstlisting}
  \end{block}

  \begin{block}{Result}
    \begin{tabular}{@{}ll@{}}
      \toprule
      \textbf{AlbumID} & \textbf{track\_count} \\
      \midrule
      1 & 2 \\
      2 & 1 \\
      3 & 2 \\
      \bottomrule
    \end{tabular}
  \end{block}

  \begin{itemize}
    \item \texttt{GROUP BY} \textbf{collapses} all rows with the same group value
          into a single output row
    \item Every column in \texttt{SELECT} must either appear in \texttt{GROUP BY}
          \textbf{or} be wrapped in an aggregate function
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{HAVING -- Filtering Groups}
  \begin{block}{Filter \emph{after} grouping}
\begin{lstlisting}[style=sqlstyle]
-- Albums with more than 1 track (HAVING filters groups)
SELECT AlbumID, COUNT(*) AS track_count
FROM Track
GROUP BY AlbumID
HAVING COUNT(*) > 1;
-- Result: AlbumID 1 (2 tracks), AlbumID 3 (2 tracks)

-- Combining WHERE (filter rows) and HAVING (filter groups)
-- Average duration of tracks >200s, per album, only albums >250s avg
SELECT AlbumID, AVG(Duration) AS avg_dur
FROM Track
WHERE Duration > 200
GROUP BY AlbumID
HAVING AVG(Duration) > 250;
\end{lstlisting}
  \end{block}

  \begin{alertblock}{WHERE vs.\ HAVING}
    \texttt{WHERE} filters individual \textbf{rows} \emph{before} grouping.
    \texttt{HAVING} filters \textbf{groups} \emph{after} grouping.
    Never put aggregate conditions in \texttt{WHERE}!
  \end{alertblock}
\end{frame}

\begin{frame}[fragile]{CASE WHEN -- Conditional Expressions}
  \begin{block}{Classify Tracks by Duration}
\begin{lstlisting}[style=sqlstyle]
SELECT Title, Duration,
  CASE
    WHEN Duration > 600  THEN 'Long (>10 min)'
    WHEN Duration >= 240 THEN 'Medium (4-10 min)'
    ELSE                      'Short (<4 min)'
  END AS length_category
FROM Track;
\end{lstlisting}
  \end{block}

  \begin{block}{Expected Result}
    \begin{tabular}{@{}lll@{}}
      \toprule
      \textbf{Title} & \textbf{Duration} & \textbf{length\_category} \\
      \midrule
      Come Together & 259  & Medium (4-10 min) \\
      Something     & 182  & Short ($<$4 min) \\
      Autobahn      & 1373 & Long ($>$10 min) \\
      Get Lucky     & 248  & Medium (4-10 min) \\
      Instant Crush & 337  & Medium (4-10 min) \\
      \bottomrule
    \end{tabular}
  \end{block}

  \begin{itemize}
    \item Conditions evaluated \textbf{in order} -- first matching \texttt{WHEN} wins
    \item If no condition matches and no \texttt{ELSE}, result is \texttt{NULL}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{CASE WHEN -- Conditional Aggregation}
  \begin{block}{Pivot-style report: count albums per era}
\begin{lstlisting}[style=sqlstyle]
SELECT
  SUM(CASE WHEN ReleaseYear < 1980 THEN 1 ELSE 0 END)
    AS before_1980,
  SUM(CASE WHEN ReleaseYear BETWEEN 1980 AND 1999 THEN 1 ELSE 0 END)
    AS the_80s_90s,
  SUM(CASE WHEN ReleaseYear >= 2000 THEN 1 ELSE 0 END)
    AS after_2000
FROM Album;
-- Result: before_1980=2, the_80s_90s=0, after_2000=1
\end{lstlisting}
  \end{block}

  \begin{examples}{Use Case}
    Conditional aggregation avoids subqueries and produces a single-row
    summary. Common in dashboards and reporting queries.
    \texttt{CASE WHEN} can appear in \texttt{SELECT}, \texttt{ORDER BY},
    and inside any aggregate function.
  \end{examples}
\end{frame}
