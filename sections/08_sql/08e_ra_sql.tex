\begin{frame}{Relational Algebra \textrightarrow{} SQL -- Unary Operators}
  \begin{center}
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}llll@{}}
    \toprule
    \textbf{Operation} & \textbf{Symbol} & \textbf{RA Syntax} & \textbf{SQL Equivalent} \\
    \midrule
    Restriction    & $\sigma$ & $\sigma_C(R)$                    & \texttt{SELECT * FROM R WHERE C} \\
    Projection     & $\pi$    & $\pi_A(R)$                       & \texttt{SELECT A FROM R} \\
    Rename (table) & $\rho$   & $\rho_N(R)$                      & \texttt{FROM R AS N} \\
    Deduplication  & $\delta$ & $\delta(\pi_A(R))$               & \texttt{SELECT DISTINCT A FROM R} \\
    Aggregation    & $\gamma$ & $\gamma_{A,\,\text{SUM}(B)}(R)$ & \texttt{SELECT A, SUM(B) \ldots GROUP BY A} \\
    Sorting        & $\tau$   & $\tau_{A\,\text{DESC}}(R)$      & \texttt{SELECT * FROM R ORDER BY A DESC} \\
    \bottomrule
  \end{tabular}
  \end{center}

  \begin{block}{Key Insight}
    $\sigma$ selects \textbf{rows} (horizontal cut); $\pi$ selects \textbf{columns} (vertical cut).
    In SQL both are expressed in a single \texttt{SELECT \ldots FROM \ldots WHERE} statement.
  \end{block}

  \begin{exampleblock}{Example -- Music Schema}
    $\pi_{\text{name, genre}}(\sigma_{\text{year} > 2000}(\text{Album}))$ \quad $\Rightarrow$ \quad
    \texttt{SELECT name, genre FROM Album WHERE year > 2000}
  \end{exampleblock}
\end{frame}

\begin{frame}{Relational Algebra \textrightarrow{} SQL -- Binary Operators}
  \begin{center}
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}lll@{}}
    \toprule
    \textbf{Operation}    & \textbf{RA Syntax}       & \textbf{SQL Equivalent} \\
    \midrule
    Union             & $R_1 \cup R_2$           & \texttt{(\ldots) UNION (\ldots)} \\
    Intersection      & $R_1 \cap R_2$           & \texttt{(\ldots) INTERSECT (\ldots)} \\
    Difference        & $R_1 \setminus R_2$      & \texttt{(\ldots) EXCEPT (\ldots)} \\
    Cartesian Product & $R_1 \times R_2$         & \texttt{SELECT * FROM R1, R2} \\
    Division          & $R_1 \div R_2$           & No direct keyword -- use \texttt{NOT EXISTS} \\
    \bottomrule
  \end{tabular}
  \end{center}

  \begin{itemize}
    \item \texttt{UNION} removes duplicates by default; \texttt{UNION ALL} keeps them (faster).
    \item \texttt{INTERSECT} / \texttt{EXCEPT} were added in MySQL only in v8.0.31.
    \item The Cartesian product (\texttt{FROM R1, R2} without a join condition) returns every
          combination of rows -- almost always unintentional in practice.
  \end{itemize}

  \begin{exampleblock}{UNION Example}
    Artists appearing on an Album \emph{or} on a Compilation: \\[2pt]
    \texttt{SELECT artist\_id FROM Album UNION SELECT artist\_id FROM Compilation}
  \end{exampleblock}
\end{frame}

\begin{frame}{Relational Algebra \textrightarrow{} SQL -- Join Operators}
  \begin{center}
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{@{}lll@{}}
    \toprule
    \textbf{Operation}   & \textbf{RA}                   & \textbf{SQL} \\
    \midrule
    Natural Join       & $R_1 \bowtie R_2$             & \texttt{R1 NATURAL JOIN R2} \\
    Inner Join         & $R_1 \bowtie_C R_2$           & \texttt{R1 INNER JOIN R2 ON C} \\
    Left Outer Join    & $R_1 \leftouterjoin R_2$      & \texttt{R1 LEFT JOIN R2 ON C} \\
    Right Outer Join   & $R_1 \rightouterjoin R_2$     & \texttt{R1 RIGHT JOIN R2 ON C} \\
    Full Outer Join    & $R_1 \fullouterjoin R_2$      & \texttt{LEFT JOIN UNION RIGHT JOIN} \\
    \bottomrule
  \end{tabular}
  \end{center}

  \begin{block}{Choosing the Right Join}
    \begin{itemize}
      \item \texttt{INNER JOIN}: only rows with a match on \textbf{both} sides.
      \item \texttt{LEFT JOIN}: all rows from the left table; unmatched right columns $=$ \texttt{NULL}.
      \item MySQL has no native \texttt{FULL OUTER JOIN} -- emulate with \texttt{LEFT JOIN UNION RIGHT JOIN}.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}[fragile]{Step-by-Step Translation: RA \textrightarrow{} SQL}
  \textbf{Task:} Find the titles of all tracks on albums released after 2010 by artist ``Aurora''.

  \medskip
  \textbf{Relational Algebra:}
  \[
    \pi_{\text{title}}
      \bigl(
        \sigma_{\text{year} > 2010}(\text{Album})
        \bowtie
        \text{Track}
        \bowtie
        \sigma_{\text{name = 'Aurora'}}(\text{Artist})
      \bigr)
  \]

  \textbf{SQL Translation:}
  \begin{lstlisting}[style=sqlstyle]
SELECT t.title
FROM   Track   AS t
JOIN   Album   AS al ON t.album_id   = al.id
JOIN   Artist  AS ar ON al.artist_id = ar.id
WHERE  ar.name = 'Aurora'
  AND  al.year > 2010;
  \end{lstlisting}

  \begin{block}{Translation Pattern}
    $\sigma$ (restriction) $\Rightarrow$ \texttt{WHERE} \quad
    $\pi$ (projection) $\Rightarrow$ \texttt{SELECT} list \quad
    $\bowtie$ (join) $\Rightarrow$ \texttt{JOIN \ldots ON}
  \end{block}
\end{frame}

\begin{frame}[fragile]{Step-by-Step Translation: Aggregation}
  \textbf{Task:} For each artist, count the number of albums they have released.

  \medskip
  \textbf{Relational Algebra:}
  \[
    \gamma_{\text{Artist.name},\; \text{COUNT}(*) \to \text{album\_count}}
      (\text{Artist} \bowtie \text{Album})
  \]

  \textbf{SQL Translation:}
  \begin{lstlisting}[style=sqlstyle]
SELECT   ar.name,
         COUNT(al.id) AS album_count
FROM     Artist AS ar
LEFT JOIN Album  AS al ON ar.id = al.artist_id
GROUP BY ar.name
HAVING   COUNT(al.id) > 0
ORDER BY album_count DESC;
  \end{lstlisting}

  \begin{itemize}
    \item $\gamma$ (aggregation) $\Rightarrow$ \texttt{GROUP BY} + aggregate function in \texttt{SELECT}.
    \item \texttt{LEFT JOIN} ensures artists with \emph{zero} albums still appear (count = 0).
    \item \texttt{HAVING} filters on aggregated values; \texttt{WHERE} cannot be used here.
  \end{itemize}
\end{frame}

\begin{frame}{SELECT Execution Order}
  SQL clauses are \textbf{written} in one order but \textbf{executed} in a different order.
  Understanding this avoids common query bugs.

  \begin{block}{Logical Execution Sequence}
    \begin{enumerate}
      \item \textbf{FROM} (and JOINs) -- identify and combine source tables
      \item \textbf{WHERE} -- filter individual rows
      \item \textbf{GROUP BY} -- collapse rows into groups
      \item \textbf{HAVING} -- filter groups (after aggregation)
      \item \textbf{SELECT} -- compute output expressions; assign column aliases
      \item \textbf{DISTINCT} -- remove duplicate rows
      \item \textbf{UNION / INTERSECT / EXCEPT} -- combine result sets
      \item \textbf{ORDER BY} -- sort the final result
      \item \textbf{LIMIT / OFFSET} -- restrict output size
    \end{enumerate}
  \end{block}

  \begin{alertblock}{Common Mistakes}
    \begin{itemize}
      \item Column aliases (assigned at step 5) \textbf{cannot} be referenced in \texttt{WHERE} (step 2)
            or \texttt{HAVING} (step 4) -- you must repeat the full expression.
      \item You \textbf{cannot} use an aggregate function directly in \texttt{WHERE} -- use \texttt{HAVING}.
    \end{itemize}
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{Execution Order -- Practical Example}
  \textbf{Task:} Find artists with more than 3 albums, sorted by album count descending.

  \begin{lstlisting}[style=sqlstyle]
SELECT   ar.name,
         COUNT(al.id) AS album_count   -- alias assigned at step 5
FROM     Artist AS ar
JOIN     Album  AS al ON ar.id = al.artist_id  -- step 1
-- WHERE album_count > 3  --> ERROR: alias unknown at step 2
GROUP BY ar.name                       -- step 3
HAVING   COUNT(al.id) > 3             -- step 4: repeat expression
ORDER BY album_count DESC;            -- step 8: alias OK here
  \end{lstlisting}

  \begin{block}{Rule of Thumb}
    \begin{itemize}
      \item Filter on \textbf{individual row values} $\Rightarrow$ \texttt{WHERE}
      \item Filter on \textbf{aggregate results} $\Rightarrow$ \texttt{HAVING}
    \end{itemize}
  \end{block}
\end{frame}
