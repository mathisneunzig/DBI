\begin{frame}[fragile]{Subqueries \textbf{(!)}}
  A \textbf{subquery} (or inner query) is a \texttt{SELECT} statement
  nested inside another SQL statement.

  \begin{block}{Three Common Positions}
\begin{lstlisting}[style=sqlstyle]
-- 1. Subquery in FROM (derived table / inline view)
SELECT name
FROM (SELECT * FROM product WHERE price > 5) AS expensive;

-- 2. Subquery in WHERE with IN
SELECT name FROM department
WHERE id IN (
  SELECT department_id FROM product
  GROUP BY department_id
  HAVING COUNT(*) > 5
);

-- 3. Correlated subquery (references outer query row by row)
SELECT name,
  (SELECT AVG(p2.price) FROM product p2
   WHERE p2.brand_id = p.brand_id) AS brand_avg
FROM product p;
\end{lstlisting}
  \end{block}

  \begin{alertblock}{Correlated Subqueries Are Slow}
    A correlated subquery is re-executed \textbf{once per row} of the outer
    query.  For large tables this can be extremely slow.  Consider rewriting
    with a \texttt{JOIN} or window function instead.
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{EXISTS, ANY, ALL \textbf{(!)}}
  \begin{block}{Set-Based Comparison Operators}
\begin{lstlisting}[style=sqlstyle]
-- EXISTS: true if subquery returns at least one row
SELECT * FROM employee AS e
WHERE EXISTS (
  SELECT 1 FROM product AS p
  WHERE p.department_id = e.department_id
);

-- ANY: true if any subquery value satisfies the condition
SELECT * FROM product
WHERE price > ANY (
  SELECT price FROM product WHERE department_id = 8
);

-- ALL: true if ALL subquery values satisfy the condition
SELECT * FROM product
WHERE price > ALL (
  SELECT price FROM product WHERE department_id = 3
);
\end{lstlisting}
  \end{block}

  \begin{itemize}
    \item \texttt{EXISTS} is often more efficient than \texttt{IN} because
          the database can stop as soon as it finds \emph{one} matching row.
    \item \texttt{> ANY} is equivalent to \texttt{> MIN(\ldots)};
          \texttt{> ALL} is equivalent to \texttt{> MAX(\ldots)}.
    \item \texttt{NOT EXISTS} is the standard way to express the
          \textbf{relational division} pattern.
  \end{itemize}
\end{frame}
