\begin{frame}[fragile]{WHERE -- Filtering Rows}
  \begin{block}{Basic Filtering -- Music Database}
\begin{lstlisting}[style=sqlstyle]
-- Albums released after 2000
SELECT Title, ReleaseYear FROM Album
WHERE ReleaseYear > 2000;

-- Artists from Germany
SELECT Name FROM Artist
WHERE Country = 'Germany';

-- Tracks shorter than 3 minutes (180 seconds)
SELECT Title, Duration FROM Track
WHERE Duration < 180;
\end{lstlisting}
  \end{block}

  \begin{block}{Key Rules}
    \begin{itemize}
      \item Only rows where the condition evaluates to \texttt{TRUE} are returned
      \item String comparisons require \textbf{single quotes}: \texttt{'Germany'}
      \item \texttt{NULL} comparisons: \textbf{never} use \texttt{= NULL},
            always use \texttt{IS NULL} / \texttt{IS NOT NULL}
    \end{itemize}
  \end{block}

  \begin{block}{Operator Precedence}
    \texttt{()} $\to$ \texttt{NOT} $\to$ \texttt{AND} $\to$ \texttt{OR}\\[2pt]
    When in doubt, add parentheses -- they cost nothing and prevent subtle bugs.
  \end{block}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{WHERE -- Logical Operators}
  \begin{block}{AND, OR, NOT -- Music Database Examples}
\begin{lstlisting}[style=sqlstyle]
-- AND: both conditions must be true
-- Electronic albums released before 2000
SELECT Title FROM Album
WHERE Genre = 'Electronic' AND ReleaseYear < 2000;
-- Result: Autobahn

-- OR: at least one condition must be true
-- Artists from UK or France
SELECT Name FROM Artist
WHERE Country = 'UK' OR Country = 'France';
-- Result: The Beatles, Daft Punk

-- Combining with parentheses (essential for correctness!)
SELECT Title FROM Album
WHERE Genre = 'Electronic'
  AND (ReleaseYear < 1980 OR ReleaseYear > 2010);
-- Result: Autobahn (1974), Random Access Memories (2013)
\end{lstlisting}
  \end{block}

  \begin{alertblock}{Common Mistake: Missing Parentheses}
    Without parentheses: \texttt{A AND B OR C} is read as \texttt{(A AND B) OR C}.
    This produces completely different results -- always use parentheses when
    mixing \texttt{AND} and \texttt{OR}.
  \end{alertblock}
\end{frame}

\begin{frame}[fragile]{WHERE -- Comparison, BETWEEN, IN}
  \begin{block}{Standard Operators and BETWEEN / IN}
\begin{lstlisting}[style=sqlstyle]
-- Standard comparison operators
SELECT Title FROM Album WHERE ReleaseYear >= 2000;
SELECT Title FROM Album WHERE Genre <> 'Rock'; -- != also works

-- BETWEEN (inclusive on both ends)
SELECT Title, Duration FROM Track
WHERE Duration BETWEEN 200 AND 350;
-- Returns: Come Together (259), Get Lucky (248), Instant Crush (337)

-- IN (equivalent to multiple OR conditions)
SELECT Name FROM Artist
WHERE Country IN ('Germany', 'France');
-- Returns: Kraftwerk, Daft Punk

-- NOT IN
SELECT Name FROM Artist
WHERE Country NOT IN ('UK');
-- Returns: Kraftwerk, Daft Punk
\end{lstlisting}
  \end{block}

  \begin{itemize}
    \item \texttt{BETWEEN} is \textbf{inclusive} on both bounds
    \item \texttt{IN} is more readable than chaining many \texttt{OR} conditions
    \item The query optimiser treats \texttt{IN} and multiple \texttt{OR}s equivalently
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{WHERE -- LIKE for Pattern Matching}
  \begin{block}{Wildcard Characters}
\begin{lstlisting}[style=sqlstyle]
-- % matches zero, one, or many characters
-- Tracks starting with 'C'
SELECT Title FROM Track WHERE Title LIKE 'C%';
-- Result: Come Together

-- Albums containing 'Road' anywhere
SELECT Title FROM Album WHERE Title LIKE '%Road%';
-- Result: Abbey Road

-- Artists ending with 'k'
SELECT Name FROM Artist WHERE Name LIKE '%k';
-- Result: Daft Punk, Kraftwerk

-- _ matches exactly one character
-- 4-letter artist names (any 4 chars)
SELECT Name FROM Artist WHERE Name LIKE '____';
\end{lstlisting}
  \end{block}

  \begin{itemize}
    \item In MySQL with the default collation, \texttt{LIKE} is \textbf{case-insensitive}
  \end{itemize}

  \begin{alertblock}{Performance: Avoid Leading Wildcards}
    \texttt{LIKE '\%term'} \textbf{cannot use an index} -- forces a full table scan.
    \texttt{LIKE 'term\%'} can use an index. For full-text search, use
    \texttt{FULLTEXT} indexes instead.
  \end{alertblock}
\end{frame}

\begin{frame}[fragile]{WHERE -- NULL Handling}
  \begin{block}{Testing for NULL}
\begin{lstlisting}[style=sqlstyle]
-- Artists with no country recorded
SELECT Name FROM Artist WHERE Country IS NULL;

-- Artists WITH a country recorded
SELECT Name FROM Artist WHERE Country IS NOT NULL;

-- NULL propagates in arithmetic
SELECT 5 + NULL;    -- Result: NULL
SELECT NULL > 0;    -- Result: NULL (not FALSE!)
\end{lstlisting}
  \end{block}

  \begin{alertblock}{\texttt{= NULL} Never Works}
    \texttt{WHERE Country = NULL} always evaluates to \texttt{NULL}
    (unknown), \textbf{never} \texttt{TRUE}. No rows are ever returned.
    This is one of the most frequent SQL beginner mistakes.
  \end{alertblock}

  \begin{block}{NULL Semantics}
    \begin{itemize}
      \item \texttt{NULL} means \emph{unknown} or \emph{not applicable} --
            it is \textbf{not} zero, empty string, or \texttt{false}
      \item Any arithmetic or comparison with \texttt{NULL} yields \texttt{NULL}
      \item SQL uses \textbf{three-valued logic}: TRUE, FALSE, UNKNOWN (NULL)
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}[fragile]{WHERE -- Common Mistakes Summary}
  \begin{alertblock}{Top 4 WHERE Mistakes}
\begin{lstlisting}[style=sqlstyle]
-- 1. NULL comparison (WRONG / CORRECT)
WHERE Country = NULL          -- always 0 rows
WHERE Country IS NULL         -- correct

-- 2. Forgetting quotes around strings
WHERE Country = Germany       -- ERROR: Germany is not a column
WHERE Country = 'Germany'     -- correct

-- 3. AND / OR precedence without parentheses
WHERE Genre = 'Rock' OR Genre = 'Electronic' AND ReleaseYear > 2000
-- Read as: Genre='Rock' OR (Genre='Electronic' AND year>2000)
-- Add parentheses to clarify intent!

-- 4. Case-sensitive string values (on some collations)
WHERE Country = 'germany'     -- may return 0 rows
WHERE Country = 'Germany'     -- correct
\end{lstlisting}
  \end{alertblock}
\end{frame}
