\subsection{Data Manipulation Language}

% ----------------------------------------------------------
\begin{frame}[fragile]{SELECT -- The Foundation}
  The \texttt{SELECT} statement retrieves data from one or more tables.
  It is the most frequently used SQL statement.

  \begin{block}{Full Syntax (clauses in writing order)}
\begin{lstlisting}[style=sqlstyle]
SELECT  column_name(s)       -- what to return
FROM    table_name           -- where to look
WHERE   condition            -- filter rows
GROUP BY column_name(s)      -- aggregate into groups
HAVING  condition            -- filter groups
ORDER BY column_name(s)      -- sort result
LIMIT   number               -- max rows to return
\end{lstlisting}
  \end{block}

  \begin{itemize}
    \item \texttt{SELECT} and \texttt{FROM} are the only \textbf{mandatory}
          clauses; all others are optional.
    \item \texttt{GROUP BY} is used together with aggregate functions;
          \texttt{HAVING} filters \emph{groups} (like \texttt{WHERE} but
          for groups, not individual rows).
  \end{itemize}

  \begin{alertblock}{Execution Order $\neq$ Writing Order}
    \texttt{FROM} $\to$ \texttt{WHERE} $\to$ \texttt{GROUP BY} $\to$
    \texttt{HAVING} $\to$ \texttt{SELECT} $\to$ \texttt{ORDER BY} $\to$
    \texttt{LIMIT}\\[2pt]
    This is why column aliases from \texttt{SELECT} cannot be used in
    \texttt{WHERE} -- \texttt{WHERE} runs before \texttt{SELECT}!
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{SELECT -- Choosing Columns}
  \begin{block}{Examples using the Music database}
\begin{lstlisting}[style=sqlstyle]
-- Return ALL columns (convenient for exploration)
SELECT * FROM Artist;

-- Return specific columns only
SELECT Name, Country FROM Artist;

-- Qualify column names (required when joining tables)
SELECT Artist.Name, Artist.Country FROM Artist;
\end{lstlisting}
  \end{block}

  \begin{block}{Result of \texttt{SELECT Name, Country FROM Artist}}
    \begin{tabular}{@{}ll@{}}
      \toprule
      \textbf{Name} & \textbf{Country} \\
      \midrule
      The Beatles & UK \\
      Kraftwerk   & Germany \\
      Daft Punk   & France \\
      \bottomrule
    \end{tabular}
  \end{block}

  \begin{alertblock}{Avoid \texttt{SELECT *} in production code}
    Column order can change when the table is altered. Transferring unneeded
    columns wastes network bandwidth. Always name the columns you need.
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{SELECT DISTINCT}
  \begin{block}{Returning Only Unique Values}
\begin{lstlisting}[style=sqlstyle]
-- All genres (with duplicates)
SELECT Genre FROM Album;
-- Returns: Rock, Electronic, Electronic

-- Unique genres only
SELECT DISTINCT Genre FROM Album;
-- Returns: Rock, Electronic

-- DISTINCT on multiple columns -- the COMBINATION must be unique
SELECT DISTINCT Genre, ReleaseYear FROM Album;
\end{lstlisting}
  \end{block}

  \begin{itemize}
    \item Without \texttt{DISTINCT}, SQL returns \textbf{all rows including duplicates}
    \item \texttt{DISTINCT} applies to the \textbf{entire row}, not just the first column
  \end{itemize}

  \begin{alertblock}{Performance Warning}
    \texttt{DISTINCT} forces the database to sort or hash the entire result set.
    On large tables this is expensive. Avoid it if your query already guarantees
    unique rows, or use \texttt{COUNT(DISTINCT col)} for counting.
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{LIMIT and OFFSET -- Pagination}
  \begin{block}{Restricting the Number of Returned Rows}
\begin{lstlisting}[style=sqlstyle]
-- Return at most 2 albums
SELECT * FROM Album LIMIT 2;

-- Skip first 2, then return the next 2 (page 2, 2 rows/page)
SELECT * FROM Album ORDER BY AlbumID LIMIT 2 OFFSET 2;

-- SQL Server syntax (for reference):
SELECT TOP 2 * FROM Album;
\end{lstlisting}
  \end{block}

  \begin{block}{Result of \texttt{SELECT Title FROM Album LIMIT 2}}
    \begin{tabular}{@{}l@{}}
      \toprule
      \textbf{Title} \\
      \midrule
      Abbey Road \\
      Autobahn \\
      \bottomrule
    \end{tabular}
  \end{block}

  \begin{alertblock}{Always combine LIMIT with ORDER BY}
    Without \texttt{ORDER BY}, the database may return a different set of rows
    each time. \texttt{LIMIT} without \texttt{ORDER BY} is only for quick
    data exploration.
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{Aliases (AS) -- Renaming Output}
  \begin{block}{Column, Table, and Expression Aliases}
\begin{lstlisting}[style=sqlstyle]
-- Column alias: rename output column
SELECT Title AS album_title, ReleaseYear AS year
FROM Album;

-- Expression alias: label a computed value
SELECT Title, Duration / 60.0 AS duration_minutes
FROM Track;

-- Table alias: shorter reference (essential for JOINs)
SELECT a.Name, al.Title
FROM Artist AS a
JOIN Album  AS al ON a.ArtistID = al.ArtistID;
\end{lstlisting}
  \end{block}

  \begin{itemize}
    \item Aliases are \textbf{temporary} -- they only exist for the duration of the query
    \item Column aliases can be used in \texttt{ORDER BY} (MySQL extension)
    \item Column aliases \textbf{cannot} be used in \texttt{WHERE} or \texttt{HAVING}
          (those clauses execute before \texttt{SELECT})
  \end{itemize}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{Aggregate Functions -- Overview}
  Aggregate functions compute a single value from a set of rows.

  \begin{block}{Common Functions -- Music Database Examples}
\begin{lstlisting}[style=sqlstyle]
-- Count all rows (including NULLs)
SELECT COUNT(*) FROM Track;            -- Result: 5

-- Count non-NULL values in a column
SELECT COUNT(Duration) FROM Track;     -- Result: 5

-- Min / Max / Sum / Average
SELECT MIN(Duration)  FROM Track;      -- Result: 182
SELECT MAX(Duration)  FROM Track;      -- Result: 1373
SELECT SUM(Duration)  FROM Track;      -- Result: 2399
SELECT AVG(Duration)  FROM Track;      -- Result: 479.8
SELECT AVG(Duration) / 60.0 AS avg_min FROM Track; -- ~8 min
\end{lstlisting}
  \end{block}

  \begin{itemize}
    \item \texttt{COUNT(*)} counts \textbf{all rows} (including \texttt{NULL});
          \texttt{COUNT(col)} counts only \textbf{non-NULL} values
    \item Most powerful when combined with \texttt{GROUP BY}
  \end{itemize}

  \begin{alertblock}{Cannot Use Aggregates in WHERE}
    Aggregates are computed \emph{after} \texttt{WHERE} filtering.
    Use \texttt{HAVING} to filter on aggregate results.
  \end{alertblock}
\end{frame}

% ----------------------------------------------------------
\begin{frame}[fragile]{Common Mistakes -- SELECT Basics}
  \begin{alertblock}{Mistake 1: Using \texttt{= NULL} instead of \texttt{IS NULL}}
\begin{lstlisting}[style=sqlstyle]
-- WRONG: always returns 0 rows (NULL = NULL is NULL, not TRUE)
SELECT * FROM Artist WHERE Country = NULL;

-- CORRECT
SELECT * FROM Artist WHERE Country IS NULL;
\end{lstlisting}
  \end{alertblock}

  \begin{alertblock}{Mistake 2: Referencing alias in WHERE}
\begin{lstlisting}[style=sqlstyle]
-- WRONG: 'dur_min' is not yet defined when WHERE runs
SELECT Duration / 60.0 AS dur_min FROM Track
WHERE dur_min > 4;

-- CORRECT: repeat the expression
SELECT Duration / 60.0 AS dur_min FROM Track
WHERE Duration / 60.0 > 4;
\end{lstlisting}
  \end{alertblock}

  \begin{alertblock}{Mistake 3: Forgetting ORDER BY with LIMIT}
\begin{lstlisting}[style=sqlstyle]
-- UNRELIABLE: could return any 3 tracks in any order
SELECT * FROM Track LIMIT 3;

-- CORRECT: deterministic result
SELECT * FROM Track ORDER BY TrackID LIMIT 3;
\end{lstlisting}
  \end{alertblock}
\end{frame}
