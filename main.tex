    \ifnum#1=1
      \begin{scope}[even odd rule]
        \clip (-0.65,0) circle (1) (0.65,0) circle (1);
        \fill[blue!30] (-0.65,0) circle (1);
      \end{scope}
    \fi
    \ifnum#2=1
      \begin{scope}
        \clip (-0.65,0) circle (1);
        \fill[blue!30] (0.65,0) circle (1);
      \end{scope}
    \fi
    \ifnum#3=1
      \begin{scope}[even odd rule]
        \clip (0.65,0) circle (1) (-0.65,0) circle (1);
        \fill[blue!30] (0.65,0) circle (1);
      \end{scope}
    \fi
    \draw[thick] (-0.65,0) circle (1);
    \draw[thick] (0.65,0) circle (1);
    \node[font=\tiny\bfseries] at (-1.5,0) {A};
    \node[font=\tiny\bfseries] at (1.5,0) {B};
  \end{tikzpicture}%
}
